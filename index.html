         <textarea id="input-place-desc" rows="3" placeholder="예: 꿔바로우 바삭하게 요청, 아이용 계란볶음밥 추천" class="w-full border rounded-lg p-2 text-sm"></textarea>

       <div id="field-speech-kr" class="space-y-1 hidden">

       </div>
       <div id="field-speech-cn" class="space-y-1 hidden">
         <label class="text-xs font-semibold text-gray-600">중국어 (한자)</label>
         <input type="text" id="input-speech-cn" placeholder="예: 不要香菜" class="w-full border rounded-lg p-2 text-sm">

         <input type="text" id="input-speech-py" placeholder="예: 부야오 씨앙차이" class="w-full border rounded-lg p-2 text-sm">

 <script>
   // 지정해주신 Firebase Realtime Database URL
   const firebaseConfig = {
     databaseURL: "https://myawesome-a2ea8-default-rtdb.firebaseio.com/"
   };
   firebase.initializeApp(firebaseConfig);
   const db = firebase.database();

   // 초기 시드 데이터 (DB가 비어있을 경우 자동 입력)
   const defaultPlaces = [
     { name: "喝丢一只鞋 (양꼬치)", tip: "💡 꿔바로우(锅包肉) 바삭하게 요청. 아이용 계란 볶음밥(鸡蛋炒饭) 추천.", query: "喝丢一只鞋" },
     { name: "Xi Ding Hai Dan Dumplings (喜鼎海胆水饺)", tip: "💡 성게알 만두 및 아이용 새우계란만두(三鲜水饺) 추천. 대기 시간 확인 필수.", query: "喜鼎海胆水饺" },
     { name: "Mangata Coffee", tip: "💡 2층 테라스 창가석이 성해대교 뷰 명당.", query: "Mangata Coffee" }
   ];

   const defaultSpeech = [
     { kr: "고수 넣지 말아주세요", cn: "不要香菜", py: "부야오 씨앙차이" },
     { kr: "덜 매운 것으로 해주세요", cn: "做辣一点吧", py: "쭤 라 이뎬 바" },
     { kr: "어린이용 의자 있나요?", cn: "有儿童椅子吗？", py: "요우 얼퉁 이쯔 마?" },
     { kr: "따뜻한 물 부탁합니다", cn: "请给我热水", py: "칭 게이 워 러슈에이" }
   ];

   let currentPlaces = {};
   let currentSpeech = {};

   // 1. Firebase 실시간 수신
   db.ref("places").on("value", (snapshot) => {
     const data = snapshot.val();
     if (!data) {
       defaultPlaces.forEach(p => db.ref("places").push(p));
       return;
     }
     currentPlaces = data;
     renderPlaces();
   });

   db.ref("speech").on("value", (snapshot) => {
     const data = snapshot.val();
     if (!data) {
       defaultSpeech.forEach(s => db.ref("speech").push(s));
       return;
     }
     currentSpeech = data;
     renderSpeech();
   });

   // 2. 렌더링 함수
   function renderPlaces() {
     const container = document.getElementById("places-container");
     container.innerHTML = "";
     Object.keys(currentPlaces).forEach(key => {
       const p = currentPlaces[key];
       const amapUrl = `https://uri.amap.com/search?keyword=${encodeURIComponent(p.query || p.name)}`;
       
       container.innerHTML += `
         <div class="bg-white p-4 rounded-xl shadow-sm border border-gray-100 space-y-2">
           <div class="flex justify-between items-start">
             <h3 class="font-bold text-gray-800 text-base">${p.name}</h3>
             <div class="flex gap-1">
               <button onclick="editItem('place', '${key}')" class="text-gray-400 hover:text-indigo-600 p-1"><i class="fa-solid fa-pen-to-square"></i></button>
               <button onclick="deleteItem('places', '${key}')" class="text-gray-400 hover:text-red-500 p-1"><i class="fa-solid fa-trash"></i></button>
             </div>
           </div>
           <p class="text-xs text-gray-600 bg-amber-50 p-2.5 rounded-lg border border-amber-100 leading-relaxed">${p.tip}</p>
           <div class="pt-1 flex justify-end">
             <a href="${amapUrl}" target="_blank" class="inline-flex items-center text-xs bg-blue-50 text-blue-600 px-2.5 py-1.5 rounded-lg font-medium hover:bg-blue-100">
               <i class="fa-solid fa-map-location-dot mr-1"></i> 고덕지도(高德地图) 연결
             </a>
           </div>
         </div>
       `;
     });
   }

   function renderSpeech() {
     const container = document.getElementById("speech-container");
     container.innerHTML = "";
     Object.keys(currentSpeech).forEach(key => {
       const s = currentSpeech[key];
       container.innerHTML += `
         <div class="bg-white p-4 rounded-xl shadow-sm border border-gray-100 flex justify-between items-center">
           <div class="space-y-1 flex-1 pr-2">
             <div class="text-xs font-semibold text-indigo-600">${s.kr}</div>
             <div class="text-lg font-bold text-gray-900">${s.cn}</div>
             <div class="text-xs text-gray-400">${s.py}</div>
           </div>
           <div class="flex items-center gap-2">
             <button onclick="speakChinese('${s.cn}')" class="w-10 h-10 bg-indigo-50 text-indigo-600 rounded-full flex items-center justify-center hover:bg-indigo-100 active:scale-95 transition">
               <i class="fa-solid fa-volume-high"></i>
             </button>
             <button onclick="editItem('speech', '${key}')" class="text-gray-300 hover:text-indigo-600 p-1"><i class="fa-solid fa-pen-to-square"></i></button>
             <button onclick="deleteItem('speech', '${key}')" class="text-gray-300 hover:text-red-500 p-1"><i class="fa-solid fa-trash"></i></button>
           </div>
         </div>
       `;
     });
   }

   // 3. 중국어 TTS
   function speakChinese(text) {
     if ('speechSynthesis' in window) {
       window.speechSynthesis.cancel();
       const utterance = new SpeechSynthesisUtterance(text);
       utterance.lang = 'zh-CN';
       utterance.rate = 0.85;
       window.speechSynthesis.speak(utterance);
     } else {
       alert("이 브라우저는 음성 재생을 지원하지 않습니다.");
     }
   }

   // 4. 모달 및 수정/삭제/추가 동작
   function switchTab(tab) {
     document.getElementById("tab-places").classList.toggle("hidden", tab !== 'places');
     document.getElementById("tab-speech").classList.toggle("hidden", tab !== 'speech');
     document.getElementById("tab-places-btn").classList.toggle("tab-active", tab === 'places');
     document.getElementById("tab-speech-btn").classList.toggle("tab-active", tab === 'speech');
   }

   function openModal(type, editId = null) {
     document.getElementById("modal-type").value = type;
     document.getElementById("modal-edit-id").value = editId || "";
     document.getElementById("modal-title").innerText = editId ? '항목 수정' : (type === 'place' ? '장소 추가' : '회화 추가');
     
     document.getElementById("field-place-name").classList.toggle("hidden", type !== 'place');
     document.getElementById("field-place-desc").classList.toggle("hidden", type !== 'place');
     document.getElementById("field-speech-kr").classList.toggle("hidden", type !== 'speech');
     document.getElementById("field-speech-cn").classList.toggle("hidden", type !== 'speech');
     document.getElementById("field-speech-py").classList.toggle("hidden", type !== 'speech');

     if (editId) {
       if (type === 'place') {
         const item = currentPlaces[editId];
         document.getElementById("input-place-title").value = item.name;
         document.getElementById("input-place-desc").value = item.tip;
       } else {
         const item = currentSpeech[editId];
         document.getElementById("input-speech-kr").value = item.kr;
         document.getElementById("input-speech-cn").value = item.cn;
         document.getElementById("input-speech-py").value = item.py;
       }
     } else {
       document.getElementById("modal-form").reset();
     }

     document.getElementById("modal").classList.remove("hidden");
   }

   function editItem(type, key) {
     openModal(type, key);
   }

   function deleteItem(path, key) {
     if (confirm("정말 삭제하시겠습니까?")) {
       db.ref(`${path}/${key}`).remove();
     }
   }

   function closeModal() {
     document.getElementById("modal").classList.add("hidden");
     document.getElementById("modal-form").reset();
   }

   document.getElementById("modal-form").addEventListener("submit", (e) => {
     e.preventDefault();
     const type = document.getElementById("modal-type").value;
     const editId = document.getElementById("modal-edit-id").value;

     if (type === 'place') {
       const name = document.getElementById("input-place-title").value;
       const tip = document.getElementById("input-place-desc").value;
       if (!name) return;

       const payload = { name, tip, query: name };
       if (editId) {
         db.ref(`places/${editId}`).update(payload);
       } else {
         db.ref("places").push(payload);
       }
     } else {
       const kr = document.getElementById("input-speech-kr").value;
       const cn = document.getElementById("input-speech-cn").value;
       const py = document.getElementById("input-speech-py").value;
       if (!cn) return;

       const payload = { kr, cn, py };
       if (editId) {
         db.ref(`speech/${editId}`).update(payload);
       } else {
         db.ref("speech").push(payload);
       }
     }

     closeModal();
   });
 </script>
